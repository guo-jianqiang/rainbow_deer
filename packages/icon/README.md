# `icon`

## install

```shell script
npm install rainbow_deer/icon
or
yarn add rainbow_deer/icon
```

## Usage

```tsx
import React, { useState } from 'react'
import Icon from '@rainbow_deer/icon'

export default () => {
  return (
    <>
      <Icon type='close' />
      <Icon type='success' />
      <Icon type='error' />
      <Icon type='warning' />
    </>
  )
}
```
